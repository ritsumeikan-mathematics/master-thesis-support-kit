    \item 変数や記号の定義（必要最小限）
    \item 最終的に求めたいもの
  \end{itemize}

  \]

\begin{frame}{提案手法の概要}
  \begin{itemize}
    \item 提案手法のアイデア（直感的な説明）
    \item 手法の流れ（アルゴリズムのステップなど）
    \item 既存手法との違い・改良点
  \end{itemize}
\end{frame}

\begin{frame}{主な理論的結果}
  \begin{itemize}
    \item 主要な定理または命題
